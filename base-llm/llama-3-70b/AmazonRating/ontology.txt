@prefix base_ontology: <https://base_ontology.com#>.
@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>.
@prefix rdfs: <http://www.w3.org/2000/01/rdf-schema#>.
@prefix owl: <http://www.w3.org/2002/07/owl#>.
@prefix xsd: <http://www.w3.org/2001/XMLSchema#>.

base_ontology:Product a owl:Class ;
    rdfs:label "Product" ;
    rdf:type owl:Class ;
    base_ontology:productId [ a rdf:Property ; 
        rdfs:domain base_ontology:Product ;
        rdfs:range xsd:string ;
        rdf:type rdf:Property ;
    ].

    rdf:type owl:Class ;

        rdfs:range xsd:integer ;
        rdf:type rdf:Property ;
    ] ;
    base_ontology:hasTimestamp [ a rdf:Property ; 
        rdfs:domain base_ontology:Review ;
        rdfs:range xsd:string ;
        rdf:type rdf:Property ;
    ].
